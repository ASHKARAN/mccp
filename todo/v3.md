## V2-1 · AST-Aware Chunking
**Files:** `crates/mccp-indexer/src/chunker.rs`
**Impact:** High — all retrieval quality depends on this

### Problem

**Step 2 — Replace the stub**

```rust
fn chunk_with_parser(&self, file: &SourceFile) -> Vec<Chunk> {
    let Some(cfg) = lang_config(file.language) else {
        return self.chunk_simple(file);
    };
    let mut parser = Parser::new();
    parser.set_language(cfg.ts_language)
          .expect("grammar version mismatch — update tree-sitter-* crates");

    let tree = match parser.parse(&file.content, None) {
        Some(t) => t,
        None    => return self.chunk_simple(file),
    };

    let mut raw_chunks = Vec::new();
    collect_splittable_nodes(
        tree.root_node(),
        &file.content,
        cfg.splittable_types,
        &mut raw_chunks,
        &file.path.to_string_lossy(),
        file.language,
    );

    if raw_chunks.is_empty() {
        return self.chunk_simple(file);
    }

    let refined = self.refine_large_chunks(raw_chunks);
    self.apply_overlap_ts_style(refined)
}
```

**Step 3 — DFS node collector**

```rust
fn collect_splittable_nodes(
    node: Node,
    src: &str,
    splittable: &[&str],
    out: &mut Vec<RawChunk>,
    path: &str,
    lang: mccp_core::Language,
) {
    if splittable.contains(&node.kind()) {
        let text = &src[node.byte_range()];
        if !text.trim().is_empty() {
            out.push(RawChunk {
                content: text.to_string(),
                start_line: node.start_position().row + 1,
                end_line:   node.end_position().row + 1,
                start_byte: node.start_byte(),
                end_byte:   node.end_byte(),
                path:       path.to_string(),
                language:   lang,
            });
        }
        // Do NOT recurse into a matched node — its children are part of this chunk
        return;
    }
    for child in node.children(&mut node.walk()) {
        collect_splittable_nodes(child, src, splittable, out, path, lang);
    }
}
```

**Step 4 — Oversized node line-split**

```rust
fn refine_large_chunks(&self, chunks: Vec<RawChunk>) -> Vec<RawChunk> {
    let max_chars = self.config.max_tokens * 4; // ~4 chars per token
    let mut out = Vec::new();
    for chunk in chunks {
        if chunk.content.len() <= max_chars {
            out.push(chunk);
        } else {
            out.extend(self.split_large_chunk_by_line(chunk, max_chars));
        }
    }
    out
}

fn split_large_chunk_by_line(&self, chunk: RawChunk, max_chars: usize) -> Vec<RawChunk> {
    let mut result = Vec::new();
    let mut current_lines: Vec<&str> = Vec::new();
    let mut current_len = 0;
    let mut start_line = chunk.start_line;

    for (i, line) in chunk.content.lines().enumerate() {
        current_len += line.len() + 1;
        current_lines.push(line);
        if current_len >= max_chars {
            result.push(RawChunk {
                content:    current_lines.join("\n"),
                start_line,
                end_line:   chunk.start_line + i,
                start_byte: 0, end_byte: 0, // approximate
                path:       chunk.path.clone(),
                language:   chunk.language,
            });
            start_line = chunk.start_line + i + 1;
            current_lines.clear();
            current_len = 0;
        }
    }
    if !current_lines.is_empty() {
        result.push(RawChunk {
            content:    current_lines.join("\n"),
            start_line,
            end_line:   chunk.end_line,
            start_byte: 0, end_byte: 0,
            path:       chunk.path.clone(),
            language:   chunk.language,
        });
    }
    result
}
```

**Step 5 — Overlap from previous tail**

```rust
fn apply_overlap_ts_style(&self, chunks: Vec<RawChunk>) -> Vec<Chunk> {
    let overlap = self.config.overlap_tokens * 4;
    let mut out = Vec::new();
    for (i, raw) in chunks.iter().enumerate() {
        let content = if i > 0 && overlap > 0 {
            let prev = &chunks[i - 1].content;
            let tail = &prev[prev.len().saturating_sub(overlap)..];
            format!("{}\n{}", tail, raw.content)
        } else {
            raw.content.clone()
        };
        out.push(Chunk::new(
            String::new(), // project_id filled by pipeline
            raw.path.clone(),
            content,
            raw.start_byte, raw.end_byte,
            raw.start_line, raw.end_line,
            ChunkScope::Method,
        ));
    }
    out
}
```

```rust
// crates/mccp-indexer/src/merkle.rs

use sha2::{Sha256, Digest};
use std::collections::BTreeMap;

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
pub struct MerkleSnapshot {
    pub root_hash:   String,
    pub file_hashes: BTreeMap<String, String>, // canonical_path → sha256
}

impl MerkleSnapshot {
    pub fn build(file_hashes: BTreeMap<String, String>) -> Self {
        let combined: String = file_hashes
            .iter()
            .map(|(p, h)| format!("{}:{}", p, h))
            .collect::<Vec<_>>()
            .join("\n");
        let root_hash = format!("{:x}", Sha256::digest(combined.as_bytes()));
        Self { root_hash, file_hashes }
    }

    /// Returns None if no changes, Some(diff) otherwise.
    pub fn diff(&self, new: &MerkleSnapshot) -> Option<SnapshotDiff> {
        if self.root_hash == new.root_hash { return None; }

        let mut added    = vec![];
        let mut removed  = vec![];
        let mut modified = vec![];

        for (path, new_hash) in &new.file_hashes {
            match self.file_hashes.get(path) {
                None                                    => added.push(path.clone()),
                Some(old) if old != new_hash            => modified.push(path.clone()),
                _                                       => {}
            }
        }
        for path in self.file_hashes.keys() {
            if !new.file_hashes.contains_key(path) { removed.push(path.clone()); }
        }
        Some(SnapshotDiff { added, removed, modified })
    }
}

#[derive(Debug, Default)]
pub struct SnapshotDiff {
    pub added:    Vec<String>,
    pub removed:  Vec<String>,
    pub modified: Vec<String>,
}

impl SnapshotDiff {
    pub fn is_empty(&self) -> bool {
        self.added.is_empty() && self.removed.is_empty() && self.modified.is_empty()
    }
    pub fn total_changed(&self) -> usize {
        self.added.len() + self.modified.len()
    }
}
```

**Step 2 — BM25 encoder**

```rust
// crates/mccp-indexer/src/bm25.rs  (new file)

use std::collections::HashMap;

pub struct BM25Encoder {
    term_to_id:   HashMap<String, u32>,
    doc_freq:     HashMap<u32, u32>,
    total_docs:   u32,
    avg_doc_len:  f32,
    k1: f32,
    b:  f32,
}

impl BM25Encoder {
    pub fn new() -> Self {
        Self { term_to_id: HashMap::new(), doc_freq: HashMap::new(),
               total_docs: 0, avg_doc_len: 0.0, k1: 1.2, b: 0.75 }
    }

    fn tokenize(text: &str) -> Vec<String> {
        text.split(|c: char| !c.is_alphanumeric() && c != '_')
            .filter(|t| t.len() > 1)
            .map(|t| t.to_lowercase())
            .collect()
    }

    pub fn register_doc(&mut self, text: &str) {
        let tokens = Self::tokenize(text);
        let n = tokens.len() as f32;
        self.avg_doc_len = (self.avg_doc_len * self.total_docs as f32 + n)
                           / (self.total_docs + 1) as f32;
        self.total_docs += 1;
        let unique: std::collections::HashSet<_> = tokens.iter().collect();
        for t in unique {
            let id = self.term_id(t.as_str());
            *self.doc_freq.entry(id).or_default() += 1;
        }
    }

    fn term_id(&mut self, term: &str) -> u32 {
        let next = self.term_to_id.len() as u32;
        *self.term_to_id.entry(term.to_string()).or_insert(next)
    }

    pub fn encode_query(&mut self, query: &str) -> Vec<(u32, f32)> {
        Self::tokenize(query)
            .into_iter()
            .filter_map(|t| {
                let id = *self.term_to_id.get(&t)?;
                let df = *self.doc_freq.get(&id)? as f32;
                let idf = ((self.total_docs as f32 - df + 0.5) / (df + 0.5) + 1.0).ln();
                Some((id, idf.max(0.0)))
            })
            .collect()
    }
}
```

```rust
// RRF fusion
fn rrf_fuse(
    a: Vec<ScoredChunk>,
    b: Vec<ScoredChunk>,
    top_k: usize,
) -> Vec<ScoredChunk> {
    const K: f64 = 60.0;
    let mut scores: HashMap<String, f64> = HashMap::new();
    let mut chunks_map: HashMap<String, ScoredChunk> = HashMap::new();

    for (rank, chunk) in a.iter().enumerate() {
        *scores.entry(chunk.chunk_id.clone()).or_default() += 1.0 / (K + rank as f64 + 1.0);
        chunks_map.entry(chunk.chunk_id.clone()).or_insert_with(|| chunk.clone());
    }
    for (rank, chunk) in b.iter().enumerate() {
        *scores.entry(chunk.chunk_id.clone()).or_default() += 1.0 / (K + rank as f64 + 1.0);
        chunks_map.entry(chunk.chunk_id.clone()).or_insert_with(|| chunk.clone());
    }

    let mut ranked: Vec<(String, f64)> = scores.into_iter().collect();
    ranked.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
    ranked.into_iter()
        .take(top_k)
        .filter_map(|(id, score)| {
            chunks_map.get(&id).map(|c| {
                let mut c = c.clone();
                c.score = score as f32;
                c
            })
        })
        .collect()
}

// In hybrid_search:
async fn hybrid_search(&self, project_id: &str, dense: &[f32], text: &str,
                        top_k: usize, filters: &Filters) -> Result<Vec<ScoredChunk>> {
    let sparse = self.bm25_encoder.lock().await.encode_query(text);
    let (dense_res, sparse_res) = tokio::join!(
        self.search_dense(project_id, dense, top_k * 2, filters),
        self.search_sparse(project_id, &sparse, top_k * 2, filters),
    );
    Ok(rrf_fuse(dense_res?, sparse_res?, top_k))
}
```

```rust
use ignore::WalkBuilder;

pub fn walk_project_files(root: &Path, extra_ignores: &[String]) -> Vec<PathBuf> {
    let global_ignore = dirs::home_dir().map(|h| h.join(".mccp").join("ignore"));

    let mut builder = WalkBuilder::new(root);
    builder
        .hidden(false)          // index dotfiles that aren't gitignored
        .git_ignore(true)
        .git_global(true)
        .git_exclude(true);

    if let Some(ref p) = global_ignore {
        if p.exists() { builder.add_ignore(p); }
    }
    for pattern in extra_ignores {
        builder.add_ignore(pattern);
    }

    builder
        .build()
        .filter_map(|entry| entry.ok())
        .filter(|e| e.file_type().map(|t| t.is_file()).unwrap_or(false))
        .filter(|e| is_supported_extension(e.path()))
        .map(|e| e.into_path())
        .collect()
}
```

```rust
// crates/mccp-providers/src/cost_tracker.rs  (new file)

use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;

/// Pricing table (USD per 1M tokens). Update as providers change.
pub fn price_per_million(provider: &str, model: &str) -> Option<f64> {
    match (provider, model) {
        ("openai",     m) if m.contains("gpt-4o")           => Some(5.00),
        ("openai",     m) if m.contains("gpt-4o-mini")      => Some(0.15),
        ("anthropic",  m) if m.contains("claude-3-5-sonnet")=> Some(3.00),
        ("anthropic",  m) if m.contains("claude-3-5-haiku") => Some(0.25),
        ("groq",       _)                                    => Some(0.05),
        ("ollama",     _) | ("vllm", _)                     => Some(0.00),
        _                                                    => None,
    }
}

#[derive(Clone)]
pub struct CostTracker {
    tokens_used:  Arc<AtomicU64>,
    spend_usd:    Arc<std::sync::Mutex<f64>>,
    max_usd:      f64,
    warn_usd:     f64,
}

impl CostTracker {
    pub fn new(max_usd: f64, warn_usd: f64) -> Self {
        Self {
            tokens_used: Arc::new(AtomicU64::new(0)),
            spend_usd:   Arc::new(std::sync::Mutex::new(0.0)),
            max_usd,
            warn_usd,
        }
    }

    /// Returns Err if the spend limit would be exceeded.
    pub fn record(&self, provider: &str, model: &str, tokens: u64) -> anyhow::Result<()> {
        self.tokens_used.fetch_add(tokens, Ordering::Relaxed);
        if let Some(price) = price_per_million(provider, model) {
            let cost = price * tokens as f64 / 1_000_000.0;
            let mut spend = self.spend_usd.lock().unwrap();
            *spend += cost;
            if *spend >= self.max_usd {
                return Err(anyhow::anyhow!(
                    "cost guard: spend limit ${:.2} reached (used ${:.4}); \
                     indexing stopped. Raise [cost_guard] max_index_cost_usd to continue.",
                    self.max_usd, *spend
                ));
            }
            if *spend >= self.warn_usd {
                tracing::warn!(spend_usd = *spend, max_usd = self.max_usd,
                    "cost guard warning: approaching spend limit");
            }
        }
        Ok(())
    }

    pub fn total_spend(&self) -> f64 { *self.spend_usd.lock().unwrap() }
    pub fn total_tokens(&self) -> u64 { self.tokens_used.load(Ordering::Relaxed) }
}
```

```rust
pub fn query_stream(
    &self,
    req: QueryRequest,
) -> impl futures::Stream<Item = Result<ScoredChunk>> {
    async_stream::try_stream! {
        let dense_vec = self.embedder.embed_one(&req.query).await?;
        let candidates = self.vector_store
            .hybrid_search(&req.project, &dense_vec, &req.query, req.top_k * 2, &req.filters)

        for chunk in self.ranker.rank_stream(candidates) {

        }
    }
```

```rust
use diffy::apply;

pub async fn apply_patch(
    state: &AppState,
    project: &str,
    path: &str,
    patch: &str,         // unified diff format
) -> Result<ApplyPatchResult> {
    let abs_path = state.resolve_path(project, path)?;

    // Security: reject path traversal
    let project_root = state.project_root(project)?;
    if !abs_path.starts_with(&project_root) {
        return Err(anyhow::anyhow!("path escapes project root"));
    }

    let original = tokio::fs::read_to_string(&abs_path).await?;
    let patched  = apply(&original, &diffy::Patch::from_str(patch)?)
        .map_err(|e| anyhow::anyhow!("patch failed: {}", e))?;

    tokio::fs::write(&abs_path, &patched).await?;

    // Trigger incremental re-index for just this file
    state.pipeline.index_file_immediate(&abs_path).await?;

    Ok(ApplyPatchResult { path: path.to_string(), bytes_written: patched.len() })
}
```

```rust
pub async fn query_cross_project(
    &self,
    req: &QueryRequest,
    allowed_projects: &[String],
) -> Result<Vec<ScoredChunk>> {
    let primary = self.query_single_project(req, &req.project).await?;

    if !req.cross_project.unwrap_or(false) { return Ok(primary); }

    let mut all = primary;
    for project in allowed_projects {
        if project == &req.project { continue; }
        let mut sub = self.query_single_project(req, project).await.unwrap_or_default();
        // Tag results with source project
        for chunk in &mut sub { chunk.source_project = Some(project.clone()); }
        all.extend(sub);
    }

    // Re-rank merged results
    all.sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap());
    all.truncate(req.top_k);
    Ok(all)
}
```

```rust
#[derive(Clone)]
pub struct AgentContext {
    pub name:       String,
    pub projects:   Vec<String>, // ["*"] means all
    pub can_write:  bool,
}

pub async fn auth_middleware(
    State(state): State<AppState>,
    mut req: Request,
    next: Next,
) -> Response {
    let token = req.headers()
        .get("Authorization")
        .and_then(|v| v.to_str().ok())
        .and_then(|v| v.strip_prefix("Bearer "));

    // If no tokens configured, allow all (single-user mode)
    if state.config.agents.is_empty() {
        req.extensions_mut().insert(AgentContext {
            name: "anonymous".into(), projects: vec!["*".into()], can_write: true,
        });
        return next.run(req).await;
    }

    match token.and_then(|t| state.config.find_agent(t)) {
        Some(agent) => {
            req.extensions_mut().insert(agent);
            next.run(req).await
        }
        None => (StatusCode::UNAUTHORIZED, "invalid agent token").into_response(),
    }
}
```
